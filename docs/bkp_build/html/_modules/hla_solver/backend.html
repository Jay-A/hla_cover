"""
hla_solver.backend module
=========================

This module provides the BackendManager class to manage parallelism strategies
(serial, thread pool, and process pool) for running the HLA greedy coverage solver.

It handles warnings about unsafe parallelism modes in Jupyter notebooks and
delegates execution to the appropriate backend implementation.

Imports:
- concurrent.futures: for thread/process pools
- tqdm: for progress bars
- warnings: for runtime warnings
- hla_solver.greedy: solver functions
- hla_solver.utils: utility functions including verbose logging
"""

import concurrent.futures
import pandas as pd
import numpy as np
from tqdm import tqdm
import warnings
from hla_solver.greedy import greedy_hla_coverage_fast
from hla_solver.utils import verbose_logger
import sys
from typing import Optional, List

def _process_partition_worker(
    df_partition,
    hla_groups,
    allele_to_id,
    allele_matrix,
    allele_id_to_positions,
    max_iter,
    top_n,
    verbose,
    partition_id
):
    """
    Execute the greedy HLA coverage algorithm on a single DataFrame partition.

    This function is designed to be top-level and picklable, so it can be used
    in both `ProcessPoolExecutor` and `ThreadPoolExecutor` for parallel execution.

    Parameters
    ----------
    df_partition : pandas.DataFrame
        A partitioned subset of the full HLA dataset.

    hla_groups : list of tuple
        List of HLA group definitions, with group name and its two allele columns.
        Format: [(group_name, col1, col2), ...]

    allele_to_id : dict
        Mapping from allele string to numeric ID, used for efficient processing.

    allele_matrix : numpy.ndarray
        Matrix representation of the allele feature space.

    allele_id_to_positions : dict
        Mapping from allele IDs to their positions within the feature matrix.

    max_iter : int
        Maximum number of iterations to perform in the greedy solver.

    top_n : int
        Number of top-scoring samples to consider at each iteration.

    verbose : bool
        If True, prints detailed output including progress and partition results.

    partition_id : int
        Integer identifier of the partition, used for logging.

    Returns
    -------
    tuple
        A 3-element tuple containing:
        
        - **selected_samples** (list):  
          Indices of samples selected by the greedy algorithm.

        - **filters** (list):  
          List of allele filters applied during selection.

        - **remainder** (pandas.DataFrame):  
          Remaining samples not covered by the selected set.

    Notes
    -----
    - This function is intended to be imported and submitted to an executor pool.
    - Logging is printed with partition ID prefix to help distinguish parallel outputs.
    - Used by `_run_thread_pool` and `_run_process_pool` in `BackendManager`.
    """

    from hla_solver.greedy import greedy_hla_coverage_fast

    selected_samples, filters, remainder = greedy_hla_coverage_fast(
        df=df_partition,
        hla_groups=hla_groups,
        allele_to_id=allele_to_id,
        allele_matrix=allele_matrix,
        allele_id_to_positions=allele_id_to_positions,
        max_iter=max_iter,
        top_n=top_n,
        verbose=verbose
    )

    if verbose:
        print(f"\n[Partition {partition_id}] Completed.")
        print(f"  ➤ Selected Samples: {selected_samples}")
        print(f"  ➤ Filters: {filters}")
        print(f"  ➤ Remainder: {len(remainder)} samples\n")

    return selected_samples, filters, remainder

    """
    Manages backend parallelism strategies for the HLA greedy solver.

    Supports three modes:

    - ``serial``: runs everything sequentially.
    - ``thread_pool``: uses a ThreadPoolExecutor for parallelism.
    - ``process_pool``: uses a ProcessPoolExecutor, with warnings if used in Jupyter notebooks.

    Parameters
    ----------
    mode : str, optional
        The parallelism mode to use. Must be one of 'serial', 'thread_pool', or 'process_pool'.
        Default is 'serial'.
    num_workers : int, optional
        Number of worker threads or processes to use in parallel modes.
        Default is 1.
    verbose : bool, optional
        If True, prints progress and debug information.
        Default is False.
    progress_mode : str, optional
        Controls progress reporting mode.
        Options: 'progress_bar', 'per_worker', 'silent'.
        Default is 'progress_bar'.

    Examples
    --------
    Create a backend manager with 4 threads and verbose output:

    >>> backend = BackendManager(mode='thread_pool', num_workers=4, verbose=True)
    >>> results = backend.run_local_global_greedy_cover(df, hla_groups, allele_to_id, allele_matrix, allele_id_to_positions)
    """

    def __init__(self, mode='serial', num_workers=1, verbose=False, progress_mode='progress_bar'):
        """
        Initialize backend manager with given configuration.

        Warns if process pool is selected inside a Jupyter notebook, as it can cause issues.

        Parameters
        ----------
        mode : str
            Parallelism mode: 'serial', 'thread_pool', or 'process_pool'.
        num_workers : int
            Number of workers to use for parallel backends.
        verbose : bool
            Enable verbose output.
        progress_mode : str
            Progress reporting mode ('progress_bar', 'per_worker', 'silent').
        """
        self.mode = mode.lower()
        self.num_workers = num_workers
        self.verbose = verbose
        self.progress_mode = progress_mode

        if self.mode == 'process_pool' and self._is_notebook():
            warnings.warn(
                "ProcessPoolExecutor is not fully safe in Jupyter notebooks. "
                "Consider using 'serial' or 'thread_pool' modes instead.",
                RuntimeWarning
            )

    def _is_notebook(self):
        """
        Detect if the code is running inside a Jupyter notebook or qtconsole.

        Returns
        -------
        bool
            True if running in a Jupyter notebook or qtconsole, False otherwise.

        Notes
        -----
        Uses the presence and type of the IPython shell to determine environment.
        """
        try:
            shell = get_ipython().__class__.__name__
            if shell == 'ZMQInteractiveShell':
                return True  # Jupyter notebook or qtconsole
            else:
                return False  # Other type (likely terminal)
        except NameError:
            return False  # Probably standard python interpreter


[docs]    def split_dataframe(self, df: pd.DataFrame, n_parts: Optional[int] = None, min_rows_per_part: int = 1000) -> List[pd.DataFrame]:
        """
        Split the DataFrame into disjoint row-wise partitions for parallel execution,
        ensuring no partition has fewer than `min_rows_per_part` rows to avoid
        excessive over-parallelization.
    
        Parameters
        ----------
        df : pandas.DataFrame
    
            The DataFrame to split.
    
        n_parts : int, optional
    
            Desired number of partitions. Defaults to self.num_workers.
    
        min_rows_per_part : int, optional
    
            Minimum number of rows per partition. Default is 1000.
    
        Returns
        -------
        List[pandas.DataFrame]
    
            List of DataFrame partitions.
        """
        n_parts = n_parts or self.num_workers
        total_rows = len(df)
    
        # Calculate max partitions so that each partition >= min_rows_per_part
        max_parts = max(1, total_rows // min_rows_per_part)
    
        # Use the minimum of requested n_parts and max_parts to avoid overparallelization
        final_parts = min(n_parts, max_parts)
    
        return np.array_split(df, final_parts)

        top_n : int
            Number of top solutions to consider in the greedy algorithm.
        top_k_global : int
            Number of top global solutions to keep.

        Returns
        -------
        tuple
            Tuple of (selected_samples, filters, remainder) from the greedy algorithm.
        """
        if self.verbose:
            print("[BackendManager] Running in SERIAL mode")

        selected_samples, filters, remainder = greedy_hla_coverage_fast(
            df=df,
            hla_groups=hla_groups,
            allele_to_id=allele_to_id,
            allele_matrix=allele_matrix,
            allele_id_to_positions=allele_id_to_positions,
            max_iter=max_iter,
            top_n=top_n,
            verbose=self.verbose
        )

        # top_k_global is not used in serial mode but could be integrated here if needed

        return selected_samples, filters, remainder

    def _run_thread_pool(
        self,
        df,
        hla_groups,
        allele_to_id,
        allele_matrix,
        allele_id_to_positions,
        max_iter,
        top_n,
        top_k_global,  # not used here
        local_bests_count=3
    ):
        """
        Greedy HLA coverage using ThreadPoolExecutor across partitions.
        At each iteration:
          - Partition remaining data
          - Each thread returns top candidates
          - Evaluate union coverage globally
          - Select best, reduce dataset, repeat
        """
        if self.verbose:
            print("[BackendManager] Running in THREADED GREEDY mode")
    
        from concurrent.futures import ThreadPoolExecutor, as_completed
        from hla_solver.selectors import (
            sample_coverage_percent,
            multi_sample_coverage_and_remainder,
            find_top_covering_samples,
        )
    
        current_df = df.copy()
        current_matrix = allele_matrix.copy()
        match_cache = {}
        selected_sample_ids = []
        filter_sequence = []
    
        for iteration in range(max_iter):
            if len(current_df) == 0:
                if self.verbose:
                    print("All samples covered.")
                break
    
            if self.verbose:
                print(f"\n[Iteration {iteration+1}] Remaining samples: {len(current_df)}")
    
            # Split dataset
            n_parts = min(self.num_workers, max(1, len(current_df) // 2500))
            partitions = self.split_dataframe(current_df, n_parts=n_parts)
            partition_matrices = np.array_split(current_matrix, n_parts)
    
            iteration_samples = []
    
            # Run threads
            with ThreadPoolExecutor(max_workers=n_parts) as executor:
                futures = [
                    executor.submit(
                        find_top_covering_samples,
                        df=df_part,
                        hla_groups=hla_groups,
                        top_n=local_bests_count,
                        allele_to_id=allele_to_id,
                        allele_matrix=mat_part,
                        allele_id_to_positions=allele_id_to_positions,
                        total_samples=len(df),
                        current_subset_indices=df_part.index.to_numpy()
                    )
                    for df_part, mat_part in zip(partitions, partition_matrices)
                ]
    
                for future in as_completed(futures):
                    top_candidates = future.result()
                    iteration_samples.extend(top_candidates)
    
            if self.verbose:
                print(f"top_candidates: {iteration_samples}")
    
            if not iteration_samples:
                if self.verbose:
                    print("No candidates returned from any partition.")
                break
    
            # Print all samples from this iteration
            if self.verbose:
                print(f"\n[Iteration {iteration+1}] Top candidates from all threads:")
                for sample in iteration_samples:
                    sample_id = sample["Sample ID"]
                    coverage, alleles = sample_coverage_percent(
                        df=current_df,  # using remaining samples
                        hla_groups=hla_groups,
                        sample_id=sample_id,
                        coverage_type="union",
                        verbose=False
                    )
                    sample["Coverage Total (%)"] = coverage  # update in-place
                    print(f"  ➤ Sample {sample_id} | Coverage: {coverage}% | Alleles: {alleles}")
    
            # Select best sample based on global coverage
            best_global_sample_id = None
            best_global_score = -1
            best_filter = None
    
            for sample in iteration_samples:
                sample_id = sample["Sample ID"]
                score = sample.get("Coverage Total (%)")
                if score is not None and score > best_global_score:
                    best_global_sample_id = sample_id
                    best_global_score = score
                    best_filter = sample["Alleles"]
    
            if best_global_sample_id is None:
                if self.verbose:
                    print("No valid global sample found.")
                break
    
            # Log best sample
            if self.verbose:
                print(f"\n[Iteration {iteration+1}] ➤ Selected Sample {best_global_sample_id} "
                      f"with coverage {best_global_score}%")
    
            # Add selection to global list
            selected_sample_ids.append(best_global_sample_id)
            filter_sequence.append(best_filter)
    
            # Global cover for this current sample
            _, _, match_cache, reduced_df = multi_sample_coverage_and_remainder(
                df=df,
                hla_groups=hla_groups,
                sample_ids=[best_global_sample_id],
                match_cache=match_cache
            )
    
            # Report updated coverage against full dataset
            _, _, _, remaining_df_full = multi_sample_coverage_and_remainder(
                df=df,
                hla_groups=hla_groups,
                sample_ids=selected_sample_ids,
                match_cache=None  # full recompute
            )
    
            if self.verbose:
                print(f"Length of reduced_df: {len(reduced_df)}")
                print(f"Length of remaining_df: {len(remaining_df_full)}")
                remaining_count = len(remaining_df_full)
                covered = len(df) - remaining_count
                print(f"[Iteration {iteration+1}] ➤ Cumulative coverage: {covered}/{len(df)} samples "
                      f"({round(100 * covered / len(df), 2)}%)")
    
            current_df = remaining_df_full
            current_matrix = allele_matrix[current_df.index]
    
        return selected_sample_ids, filter_sequence, current_df
        

    def _run_process_pool(
        self, df, hla_groups, allele_to_id, allele_matrix, allele_id_to_positions,
        max_iter, top_n, top_k_global
    ):
        """
        Run the greedy coverage algorithm using a process pool.
    
        This method partitions the input DataFrame into `num_workers` parts,
        then applies the `greedy_hla_coverage_fast` solver to each partition
        in parallel using Python's `ProcessPoolExecutor`. Each partition is
        handled by the `_process_partition_worker` function.
    
        This approach enables data-parallel processing on multi-core CPUs,
        improving performance on large datasets (outside Jupyter environments).
    
        Parameters
        ----------
        df : pandas.DataFrame
            Input DataFrame containing HLA allele information.
        
        hla_groups : list of tuple
            List of tuples defining HLA groupings and corresponding column names.
            Format: [(group_name, col1, col2), ...]
        
        allele_to_id : dict
            Mapping of allele names to numeric IDs used in the allele matrix.
    
        allele_matrix : np.ndarray
            Matrix representation of alleles or allele features (e.g., binary presence matrix).
        
        allele_id_to_positions : dict
            Mapping from allele IDs to positions or indices within the allele matrix.
        
        max_iter : int
            Maximum number of iterations for the greedy solver per partition.
    
        top_n : int
            Number of top candidates to consider at each iteration of the greedy algorithm.
    
        top_k_global : int
            Currently unused in process pool mode, but retained for compatibility with interface.
    
        Returns
        -------
        list of tuple
            A list of tuples, each corresponding to one partition:
            (selected_samples, filters, remainder)
    
        Notes
        -----
        - Process pool parallelism may be unstable in Jupyter environments due to pickling issues.
          Use `_run_thread_pool` in those cases.
        - `_process_partition_worker` is defined at the module top-level to ensure it is picklable.
        - This function does not perform result aggregation across partitions.
        """
        if self.verbose:
            print("[BackendManager] Running in PROCESS POOL mode")
    
        partitions = self.split_dataframe(df, n_parts=self.num_workers)
    
        from concurrent.futures import ProcessPoolExecutor, as_completed
    
        results = []
        with ProcessPoolExecutor(max_workers=self.num_workers) as executor:
            future_to_pid = {
                executor.submit(
                    _process_partition_worker,
                    part,
                    hla_groups,
                    allele_to_id,
                    allele_matrix,
                    allele_id_to_positions,
                    max_iter,
                    top_n,
                    self.verbose,
                    i
                ): i for i, part in enumerate(partitions)
            }
    
            for future in as_completed(future_to_pid):
                pid = future_to_pid[future]
                try:
                    result = future.result()
                    results.append(result)
                except Exception as e:
                    print(f"[Partition {pid}] Failed with error: {e}")
    
        return results  # list of tuples: (selected_samples, filters, remainder)


[docs]    def run_local_global_greedy_cover(
        self,
        df,
        hla_groups,
        allele_to_id,
        allele_matrix,
        allele_id_to_positions,
        max_iter=10,
        top_n=1,
        top_k_global=5,
    ):
        """
        Run the local-global greedy coverage algorithm using the selected backend.

        Dispatches the call to the appropriate internal method based on the `mode`.

        Parameters
        ----------
        df : pandas.DataFrame
            Input data frame containing allele information.
        hla_groups : list
            Groups of HLA alleles to cover.
        allele_to_id : dict
            Mapping from allele name to numeric ID.
        allele_matrix : numpy.ndarray or similar
            Matrix representing alleles and their features.
        allele_id_to_positions : dict
            Maps allele IDs to positions in the matrix or dataset.
        max_iter : int, optional
            Maximum iterations for the greedy algorithm.
            Default is 10.
        top_n : int, optional
            Number of top solutions to consider in the greedy algorithm.
            Default is 1.
        top_k_global : int, optional
            Number of top global solutions to keep.
            Default is 5.

        Returns
        -------
        tuple
            A tuple of (selected_samples, filters, remainder), as returned by the backend algorithm.

        Raises
        ------
        ValueError
            If an unknown backend mode is specified.
        """
        if self.mode == 'serial':
            print( "\nRunning in serial mode." )
            return self._run_serial(
                df, hla_groups, allele_to_id, allele_matrix, allele_id_to_positions,
                max_iter, top_n, top_k_global
            )
        elif self.mode == 'thread_pool':
            print( "\nRunning in thread_pool mode." )
            return self._run_thread_pool(
                df, hla_groups, allele_to_id, allele_matrix, allele_id_to_positions,
                max_iter, top_n, top_k_global
            )
        elif self.mode == 'process_pool':
            print( "\nRunning in process_pool mode." )
            return self._run_process_pool(
                df, hla_groups, allele_to_id, allele_matrix, allele_id_to_positions,
                max_iter, top_n, top_k_global
            )
        else:
            raise ValueError(f"Unknown backend mode: {self.mode}")
